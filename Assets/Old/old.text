  <script>
    /* ─── Room bounds: keeps camera inside ±10 walls ─── */
    AFRAME.registerComponent('room-bounds', {
      tick: function () {
        const el  = this.el;
        const pos = el.object3D.getWorldPosition(new THREE.Vector3());

        const margin = 0.5;
        const minX = -10 + margin, maxX = 10 - margin;
        const minZ = -10 + margin, maxZ = 10 - margin;

        const newX = Math.min(Math.max(pos.x, minX), maxX);
        const newZ = Math.min(Math.max(pos.z, minZ), maxZ);

        if (newX !== pos.x || newZ !== pos.z) {
          const parent    = el.parentEl;
          const parentPos = parent.getAttribute('position');
          parent.setAttribute('position', {
            x: parentPos.x + (newX - pos.x),
            y: parentPos.y,
            z: parentPos.z + (newZ - pos.z)
          });
        }
      }
    });
  </script>
